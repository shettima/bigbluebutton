001    package org.liquibase.grails;
002    
003    import liquibase.xml.XmlWriter;
004    import org.w3c.dom.Document;
005    import com.sun.org.apache.xml.internal.serialize.XMLSerializer;
006    import com.sun.org.apache.xml.internal.serialize.OutputFormat;
007    
008    import java.io.OutputStream;
009    import java.io.IOException;
010    
011    public class GrailsXmlWriter implements XmlWriter {
012    
013        public void write(Document doc, OutputStream outputStream) throws IOException {
014            OutputFormat format = new OutputFormat(doc);
015            format.setIndenting(true);
016            XMLSerializer serializer = new XMLSerializer(outputStream, format);
017            serializer.asDOMSerializer();
018            serializer.serialize(doc);
019    
020        }
021    }
